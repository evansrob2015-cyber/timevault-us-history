async function performVideoSearch() { const query = videoSearchInput.value; const curated = filterVideos(query); // keep curated as default let combined = curated; // OPTIONAL: enable live search when query is specific enough if (query.trim().length >= 6) { try { const live = await searchYouTubeBackend(query); combined = curated.concat(live); } catch (e) { console.warn('Live YouTube search failed', e); } } renderVideoResults(combined); if (combined.length) { setActiveVideo(combined[0]); } else { videoPlayer.src = ''; videoViewerTitle.textContent = 'No match yet.'; videoViewerDetails.textContent = 'Try a broader topic. Curated and live search both came up empty.'; videoViewerTags.innerHTML = ''; } }
